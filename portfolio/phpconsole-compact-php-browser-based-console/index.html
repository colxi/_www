PHPConsole: Browser PHP console

PHPConsole is compact browser-based application, wich allows you to interact with the server-side PHP interpreter in live, thanks to the implemented AJAX communication system. It’s lightweight (less than 10KB) and single file distribution makes it quite portable. Doesn’t claim to be the most complete brower-based PHP console implementation, but is enough powerfull for fast tests and code snippets debugging.

Implemented macros:
– ls([path])
Outputs the provided dirpath contents (or current dir if no argument is pased).
 
– cat(filename)
Outputs the content of the requested file.
 
– help()
List the available macros.
 
** Macros is a 1.5v new feature. In future releases more macros should be available.
 
Usage:
Upload file to remote ftp space, and run page from browser. Login on page load, and use CTRL+ENTER for code execution.
 
Warning:
Ensure to change the default password before uploading it!
 
To do:
Additional features, like file uploading,  line numbers in code editor,  remote filesystem management are in mind for future releases.

        function checkTab(evt) {
            var tab = "    ";
            var t = evt.target;
            var ss = t.selectionStart;
            var se = t.selectionEnd;
            // Tab key - insert tab expansion
            if (evt.keyCode == 9) {
                evt.preventDefault();
                // Special case of multi line selection
                if (ss != se && t.value.slice(ss,se).indexOf("n") != -1) {
                    // In case selection was not of entire lines (e.g. selection begins in the middle of a line)
                    // we ought to tab at the beginning as well as at the start of every following line.
                    var pre = t.value.slice(0,ss);
                    var sel = t.value.slice(ss,se).replace(/n/g,"n"+tab);
                    var post = t.value.slice(se,t.value.length);
                    t.value = pre.concat(tab).concat(sel).concat(post);
                           
                    t.selectionStart = ss + tab.length;
                    t.selectionEnd = se + tab.length;
                }
                // "Normal" case (no selection or selection on one line only)
                else {
                    t.value = t.value.slice(0,ss).concat(tab).concat(t.value.slice(ss,t.value.length));
                    if (ss == se) {
                        t.selectionStart = t.selectionEnd = ss + tab.length;
                    }
                    else {
                        t.selectionStart = ss + tab.length;
                        t.selectionEnd = se + tab.length;
                    };
                };
            };
